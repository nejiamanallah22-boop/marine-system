function toggleUser(userId, enable) { let users = JSON.parse(localStorage.getItem(USER_KEY) || "[]"); let user = users.find(u => u.id === userId); if(user) { user.enabled = enable; localStorage.setItem(USER_KEY, JSON.stringify(users)); renderUsers(); showToast(enable ? "✅ تم تفعيل المستخدم" : "⚠️ تم تعطيل المستخدم"); } } function deleteUser(userId) { if(confirm("هل أنت متأكد من حذف هذا المستخدم؟")) { let users = JSON.parse(localStorage.getItem(USER_KEY) || "[]"); users = users.filter(u => u.id !== userId); localStorage.setItem(USER_KEY, JSON.stringify(users)); renderUsers(); showToast("✅ تم حذف المستخدم"); } } // ==================== دوال التنقل بين الصفحات ==================== function showPage(page) { const pages = ['pageMain', 'pageMaint', 'pageEff', 'pageSupport', 'pageTrack', 'pageUsers']; pages.forEach(p => document.getElementById(p).classList.add('hidden')); if(page === 'main') document.getElementById('pageMain').classList.remove('hidden'); else if(page === 'maint') { document.getElementById('pageMaint').classList.remove('hidden'); renderMaint(); } else if(page === 'eff') { document.getElementById('pageEff').classList.remove('hidden'); renderEff(); } else if(page === 'support') { document.getElementById('pageSupport').classList.remove('hidden'); renderTickets(); } else if(page === 'track') { if(canManageUsers()) { document.getElementById('pageTrack').classList.remove('hidden'); renderTrack(); } else showToast("غير مسموح - هذه الصفحة للمسؤول فقط", true); } else if(page === 'users') { if(canManageUsers()) { document.getElementById('pageUsers').classList.remove('hidden'); renderUsers(); } else showToast("غير مسموح - هذه الصفحة للمسؤول فقط", true); }
