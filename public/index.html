function showPage(page) { const pages = ['pageMain','pageMaint','pageEff','pageSupport','pageTrack','pageUsers']; pages.forEach(p => document.getElementById(p).classList.add('hidden')); if (page === 'main') { document.getElementById('pageMain').classList.remove('hidden'); renderMain(); } else if (page === 'maint') { document.getElementById('pageMaint').classList.remove('hidden'); renderMaint(); } else if (page === 'eff') { document.getElementById('pageEff').classList.remove('hidden'); renderEff(); } else if (page === 'support') { document.getElementById('pageSupport').classList.remove('hidden'); renderTickets(); } else if (page === 'track') { if (canManageUsers()) { document.getElementById('pageTrack').classList.remove('hidden'); renderTrack(); } else { showToast("غير مصرح لك بالوصول إلى هذه الصفحة", true); document.getElementById('pageMain').classList.remove('hidden'); } } else if (page === 'users') { if (canManageUsers()) { document.getElementById('pageUsers').classList.remove('hidden'); renderUsers(); } else { showToast("غير مصرح لك بالوصول إلى هذه الصفحة", true); document.getElementById('pageMain').classList.remove('hidden'); } } }
